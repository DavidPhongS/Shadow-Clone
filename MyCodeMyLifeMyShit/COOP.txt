class Student
{
	//Atributy
	public string Jmeno { get; set; }
	public int Vek { get; set; }
	public string Trida { get; set; }
	public List<int> Znamky { get; set; }

	//KONSTRUKTOR:
	public Student(string jmeno, int vek, string trida)
	{
		Jmeno = jmeno;
		Vek = vek;
		Trida = trida;
		Znamky = new List<int>();

	}

	public void PridejZnamku(int znamka)
	{
		Znamky.Add(znamka);
	}

	public void VypisInfo()
	{
		Console.WriteLine ("Jmeno: " + Jmeno);
		Console.WriteLine("Vek: " + Vek);
		Console.WriteLine("Trida: " + Trida);
		if (Znamky.Count > 0)
		{
			foreach (int z in Znamky)
			{
				Console.WriteLine(" " + z);
			}
		}
	}
}

class Program
{
	static void Main()
	{
		Console.WriteLine("Kolik mas penize na bezny ucet?");
		double zustatekBU = double.Parse(Console.ReadLine());

		while (zustatekBU < 0)
		{
			Console.WriteLine("Najdi praci");
			Console.WriteLine("Kolik mas penize na bezny ucet?");
			zustatekBU = double.Parse(Console.ReadLine());
		}

		Console.WriteLine("Kolik mas penize na sporici ucet?");
		double zustatekSU = double.Parse(Console.ReadLine());
		while (zustatekSU < 0)
		{
			Console.WriteLine("Cislo je negativni");
			Console.WriteLine("Kolik mas penize na sporici ucet?");
			zustatekSU = double.Parse(Console.ReadLine());
		}


		Bank michael = new Bank(zustatekBU, zustatekSU);

		bool aktivace = true;
		while (aktivace)
		{
			Console.WriteLine("Co chces delat?\n(V)ybrat penize\n(S)ebrat penize\n(Z)obrazit informaci\n(H)istorie\n(U)rok simulace\n(K)onec");
			char halo = char.ToUpper(char.Parse(Console.ReadLine()));

			switch (halo)
			{
				case 'V':
					Console.WriteLine("Zadej kolik penez chces pridat: ");
					double pridat = double.Parse(Console.ReadLine());
					michael.PridatPenize(pridat);
					break;

				case 'S':
					Console.WriteLine("Zadej kolik penez chces sebrat: ");
					double sebrat = double.Parse(Console.ReadLine());
					michael.SebratPenize(sebrat);
					break;

				case 'Z':
					michael.ZobrazitInformace();
					break;

				case 'H':
					michael.Historie();
					break;

				case 'U':
					Console.WriteLine("Zadej kolik let: ");
					double urok = double.Parse(Console.ReadLine());
					michael.Urok(urok);
					break;

				case 'K':
					Console.WriteLine("Ukončuji program. Nashledanou!");
					aktivace = false;
					break;

				default:
					Console.WriteLine("Neplatná volba, zkus to znovu.");
					break;
			}
		}
	}
}

class Bank
{
	double urok = 4.3;
	public double ZustatekBU { get; set; }
	public double ZustatekSU { get; set; }
	public List<string> BankList { get; set; } = new List<string>();

	public Bank(double zustatekBU, double zustatekSU)
	{
		ZustatekBU = zustatekBU;
		ZustatekSU = zustatekSU;
	}

	public void PridatPenize(double penize)
	{
		ZustatekBU += penize;
		Console.WriteLine("\nPenize: \n" + ZustatekBU);
		BankList.Add($"+{penize}");
	}
	public void SebratPenize(double penize)
	{
		ZustatekBU -= penize;
		Console.WriteLine("\nPenize: \n" + ZustatekBU);
		BankList.Add($"-{penize}");
	}

	public void ZobrazitInformace()
	{
		Console.WriteLine($"\nPenize v BU: {ZustatekBU}\nPenize v SU: {ZustatekSU}\n");
	}

	public void Historie()
	{
		foreach (string n in BankList)
		{
			Console.WriteLine($"\n{n}\n");
		}
	}

	public void Urok(double rok)
	{
		double estimate = ((ZustatekBU / 100) * urok) + ZustatekBU;
		Console.WriteLine($"Urok % je: {urok}");
		for (int i = 1; i <= rok; i++)
		{
			Console.WriteLine($"{i}. rok je: {estimate}");
			estimate = ((estimate / 100) * urok) + estimate;
		}
	}
}
