4. Random number between max and min:
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
	srand(time(NULL));
	int max, min, random;
	printf("Zadej max: ");
	scanf_s("%d", &max);

	printf("Zadej min: ");
	scanf_s("%d", &min);

	int rozdil = max - min + 1;  //Example: min la 3 max la 9 rozdil = 9-3+1 = 8 

	for (int i = 0; i < 10; i++) {
		random = rand() % rozdil + min; // Generate tu 0 - 8, + min = 3 la 3 - 8
			printf("Vysledek: %d\n", random);
	}
	return 0;
}

10. Palindrom string:
#include <stdio.h>
#include <string.h>

int main() {
	char string[16], string2[16];
	printf("Zadej string pls (max 15): ");
	scanf_s("%15s", string, 15);

	int delka = strlen(string);

	for (int i = 0; i < delka; i++) {
		string2[i] = string[delka - i - 1];
	}

	string2[delka] = '\0';
	printf("\n%s", string2);

	if (strcmp(string, string2) == 0) {
		printf("\n\nJe to palindrom\n");
	}
	else {
		printf("\n\nNeni to palindrom\n");
	}
	return 0;
}
