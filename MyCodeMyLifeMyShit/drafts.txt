4. Random number between max and min:
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
	srand(time(NULL));
	int max, min, random;
	printf("Zadej max: ");
	scanf_s("%d", &max);

	printf("Zadej min: ");
	scanf_s("%d", &min);

	int rozdil = max - min + 1;  //Example: min la 3 max la 9 rozdil = 9-3+1 = 8 

	for (int i = 0; i < 10; i++) {
		random = rand() % rozdil + min; // Generate tu 0 - 8, + min = 3 la 3 - 8
			printf("Vysledek: %d\n", random);
	}
	return 0;
}

10. Palindrom string:
#include <stdio.h>
#include <string.h>

int main() {
	char string[16], string2[16];
	printf("Zadej string pls (max 15): ");
	scanf_s("%15s", string, 15);

	int delka = strlen(string);

	for (int i = 0; i < delka; i++) {
		string2[i] = string[delka - i - 1];
	}

	string2[delka] = '\0';
	printf("\n%s", string2);

	if (strcmp(string, string2) == 0) {
		printf("\n\nJe to palindrom\n");
	}
	else {
		printf("\n\nNeni to palindrom\n");
	}
	return 0;
}

12. Check password:
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
	char string[51];
	printf("Zadej string pls diky (max 50 btw): ");
	scanf_s("%50s", &string, 50);

	int delka = strlen(string);
	int mala = 0, velka = 0, cislo = 0, znak = 0;

	for (int i = 0; i < delka; i++) {
		if (string[i] >= 'a' && string[i] <= 'z') {
			mala++;
		}
		else if (string[i] >= 'A' && string[i] <= 'Z') {
			velka++;
		}
		else if (string[i] >= '0' && string[i] <= '9') {
			cislo++;
		}
		else if (string[i] >= '!' && string[i] <= '/') {
			znak++;
		}
	}

	printf("\n%d mala, %d velka, %d cislo, %d znak\n", mala, velka, cislo, znak);

	if (mala == 3 && velka == 2 && cislo == 1 && znak == 1) {
		printf("\nKrasne heslo\n");
	}
	else {
		printf("\nNehezke heslo\n");
	}
	return 0;
}

19. Pyramid:
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>

void pyramida(int patra) {
	for (int i = 1; i <= patra; i++) {
		for (int j = 0; j < patra - i; j++) {
			printf(" "); // vypise mezer zleva
		}
		for (int j = 0; j < (2 * i) - 1; j++) {
			printf("#"); // (2 * 1) znamena ze vypise doleva i doprava znaky. -1 aby mohl zustat uprostred
		}
		printf("\n"); // vyska
	}
}
int main() {
	pyramida(10);
	return 0;
}

int fuk(int n) {
	if (n == 0)
		return 1;
	return n * fuk(n - 1);
}
