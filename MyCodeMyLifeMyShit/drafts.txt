1. RNG:
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main (){
	srand(time(NULL)); //srand(time(NULL)) will generates number number depends on UTE (time), without (time(NULL)) the number will stays the same

							  //if put 100+1 then the number will be between 1-100
	int cislo = rand() % 100; //rand will then use the number the seed gives and then gives the random number between 0-100 

	printf("Nahodne cislo: %d", cislo);
	return 0;
}

4. Random number between max and min:
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {
	srand(time(NULL));
	int max, min, random;
	printf("Zadej max: ");
	scanf_s("%d", &max);

	printf("Zadej min: ");
	scanf_s("%d", &min);

	int rozdil = max - min + 1;  //Example: min la 3 max la 9 rozdil = 9-3+1 = 8 

	for (int i = 0; i < 10; i++) {
		random = rand() % rozdil + min; // Generate tu 0 - 8, + min = 3 la 3 - 8
			printf("Vysledek: %d\n", random);
	}
	return 0;
}
